#Zad2

```c
#include<stdio.h>

int ile_razy(int n){
scanf("%d",&n);
return n;}

void f2(int n){
int i;
for (i=1;i<=n;i++)
printf("Dzien dobry\n");}

int main(){
int n;
printf("Podaj n: ");
n=ile_razy(n);
f2(n);
return 0;
}
```
